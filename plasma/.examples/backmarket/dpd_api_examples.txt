Script that keeps geosession etc:

#!/bin/bash
# dpd_shipment.sh
# One-shot DPD Local API script: login → create shipment → fetch label (PDF + ZPL)

# --- Credentials ---
USERNAME="S4D"
PASSWORD="21SCREAM"
ACCOUNT="2132450"   # pulled from your geoSession token (this is your numeric account id)

# --- Shipment details ---
REF="ORDER12345"
SHIP_DATE="$(date -u +"%Y-%m-%dT09:00:00")"   # collection today 9am UTC
RECIP_NAME="Barry Smith"
RECIP_TEL="07928798705"
RECIP_EMAIL="pcouzens83@gmail.com"
RECIP_ADDR_PROPERTY="5"
RECIP_ADDR_STREET="Checkley Grove"
RECIP_ADDR_TOWN="Stoke-on-Trent"
RECIP_ADDR_COUNTY="Staffordshire"
RECIP_ADDR_PC="ST3 5TE"

DESC="Apple Mobile Phone"
VALUE=700
WEIGHT=1.0
SERVICE="2^12"   # DPD Local Next Day

# --- Step 1: Login ---
echo "Logging in..."
AUTH=$(echo -n "${USERNAME}:${PASSWORD}" | base64)
LOGIN_RESPONSE=$(curl -s -X POST "https://api.dpdlocal.co.uk/user/?action=login" \
  -H "Authorization: Basic ${AUTH}" \
  -H "Content-Type: application/json" \
  -H "Accept: application/json")

GEOSESSION=$(echo "$LOGIN_RESPONSE" | jq -r '.data.geoSession')
if [ "$GEOSESSION" = "null" ]; then
  echo "❌ Login failed: $LOGIN_RESPONSE"
  exit 1
fi
echo "✅ Logged in. GeoSession: $GEOSESSION"

# --- Step 2: Create Shipment ---
echo "Creating shipment..."
CREATE_PAYLOAD=$(jq -n \
  --arg ref "$REF" \
  --arg date "$SHIP_DATE" \
  --arg name "$RECIP_NAME" \
  --arg tel "$RECIP_TEL" \
  --arg email "$RECIP_EMAIL" \
  --arg property "$RECIP_ADDR_PROPERTY" \
  --arg street "$RECIP_ADDR_STREET" \
  --arg town "$RECIP_ADDR_TOWN" \
  --arg county "$RECIP_ADDR_COUNTY" \
  --arg pc "$RECIP_ADDR_PC" \
  --arg desc "$DESC" \
  --argjson value $VALUE \
  --argjson weight $WEIGHT \
  --arg service "$SERVICE" \
  '{
    collectionDate: $date,
    consignment: [{
      consignmentRef: $ref,
      collectionDetails: {
        contactDetails: { contactName: "S4D Warehouse", telephone: "01782 000000" },
        address: {
          organisation: "S4D Ltd",
          countryCode: "GB",
          postcode: "ST3 5XA",
          street: "Parkhall road",
          town: "Stoke on Trent",
          county: "Staffordshire"
        }
      },
      deliveryDetails: {
        contactDetails: { contactName: $name, telephone: $tel },
        address: {
          property: $property,
          street: $street,
          town: $town,
          county: $county,
          postcode: $pc,
          countryCode: "GB"
        },
        notificationDetails: { email: $email, mobile: $tel }
      },
      networkCode: $service,
      numberOfParcels: 1,
      totalWeight: $weight,
      parcelDescription: $desc,
      liability: true,
      liabilityValue: $value
    }]
  }')

SHIPMENT_RESPONSE=$(curl -s -X POST "https://api.dpdlocal.co.uk/shipping/shipment" \
  -H "Content-Type: application/json" \
  -H "Accept: application/json" \
  -H "GeoClient: account/${ACCOUNT}" \
  -H "GeoSession: ${GEOSESSION}" \
  -d "$CREATE_PAYLOAD")

SHIPMENT_ID=$(echo "$SHIPMENT_RESPONSE" | jq -r '.data.shipmentId')
CONSIGNMENT_NO=$(echo "$SHIPMENT_RESPONSE" | jq -r '.data.consignmentDetail[0].consignmentNumber')

if [ "$SHIPMENT_ID" = "null" ]; then
  echo "❌ Shipment creation failed: $SHIPMENT_RESPONSE"
  exit 1
fi
echo "✅ Shipment created. ID: $SHIPMENT_ID  Consignment: $CONSIGNMENT_NO"

# --- Step 3a: Get PDF Label ---
echo "Fetching PDF label..."
curl -s -X GET "https://api.dpdlocal.co.uk/shipping/labels?shipmentId=${SHIPMENT_ID}&labelFormat=PDF" \
  -H "GeoClient: account/${ACCOUNT}" \
  -H "GeoSession: ${GEOSESSION}" \
  -o label.pdf

if [ -s label.pdf ]; then
  echo "✅ PDF label saved as label.pdf"
else
  echo "❌ PDF label not generated"
fi

# --- Step 3b: Get ZPL Label ---
echo "Fetching ZPL label..."
curl -s -X GET "https://api.dpdlocal.co.uk/shipping/labels?shipmentId=${SHIPMENT_ID}&labelFormat=ZPL" \
  -H "GeoClient: account/${ACCOUNT}" \
  -H "GeoSession: ${GEOSESSION}" \
  -o label.zpl

if [ -s label.zpl ]; then
  echo "✅ ZPL label saved as label.zpl"
else
  echo "❌ ZPL label not generated"
Fi


# Step 1. Log in, save cookies
curl -c cookies.txt \
  -X POST "https://apis.myshipments.dpdlocal.co.uk/auth/login" \
  -H "Content-Type: application/json" \
  -d '{"email":"S4D@dpdlocal.dpduk.live","password":"21SCREAM"}'

#Step2

curl -b cookies.txt \
  "https://apis.myshipments.dpdlocal.co.uk/shipping/shipments/{$shipment_id}/labels?parcelNumbers={$parcel_id}&printerType=1" \
  -o label.zpl

Where shipment_id and parcel_id are the values we saved in tbl_orders.parcel_id/shipment_id
